Imports System.Data.SqlClient
Public Class Form1
    Private Sub Button1_Click(sender As Object, e As EventArgs) Handles Button1.Click
        Dim connetionString As String
        Dim cnn As SqlConnection
        Dim cmd As SqlCommand
        Dim sql As String

        connetionString = "Server=localhost;Database=students;Trusted_Connection=True;"
        sql = "insert into dbo.students values (1,'hghg',78)"

        cnn = New SqlConnection(connetionString)
        Try
            cnn.Open()
            cmd = New SqlCommand(sql, cnn)
            cmd.ExecuteNonQuery()
            cmd.Dispose()
            cnn.Close()
            MsgBox(" ExecuteNonQuery in SqlCommand executed !!")
        Catch ex As Exception
            MsgBox("Can not open connection ! ")
        End Try
    End Sub

    Private Sub Button2_Click(sender As Object, e As EventArgs) Handles Button2.Click
        Dim connetionString As String
        Dim cnn As SqlConnection
        Dim cmd As SqlCommand
        Dim sql As String
        Dim str As String
        Dim reader As SqlDataReader

        connetionString = "Server=localhost;Database=students;Trusted_Connection=True;"
        sql = "select * from students"
        TextBox1.Text = ""
        cnn = New SqlConnection(connetionString)
        Try
            cnn.Open()
            cmd = New SqlCommand(sql, cnn)
            reader = cmd.ExecuteReader()
            While reader.Read()
                str = str & reader.Item(0) & "  -  " & reader.Item(1) & "  -  " & reader.Item(2) & vbCrLf

            End While

            reader.Close()
            cmd.Dispose()
            cnn.Close()
            TextBox4.Text = str
        Catch ex As Exception
            MsgBox("Can not open connection ! ")
        End Try
    End Sub
End Class
